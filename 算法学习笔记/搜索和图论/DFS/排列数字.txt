#include<iostream>

using namespace std;

const int N = 10;

int path[N];
bool state[N];
int n;

int dfs(int u)
{
    if(u > n)//数字填完则输出
    {
        for(int i = 1; i <= n; i++)
            cout<< path[i] << ' ';
        cout << endl;
    }
    
    for(int i = 1; i <= n; i++)//空位上可以选择1~n
    {
        if(!state[i])
        {
            path[u] = i;
            state[i] = false;
            dfs(u+1);
            state[i] = true;
        }
    }
}

int main()
{
    cin>>n;
    dfs(1);
    return 0;
}
